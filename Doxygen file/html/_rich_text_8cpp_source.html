255  std::vector<sf::String> explode(const sf::String& string, sf::Uint32 delimiter)
256  {
257  if (string.isEmpty())
258  return std::vector<sf::String>();
259 
260  // For each character in the string
261  std::vector<sf::String> result;
262  sf::String buffer;
263  for (sf::Uint32 character : string) {
264  // If we've hit the delimiter character
265  if (character == delimiter) {
266  // Add them to the result vector
267  result.push_back(buffer);
268  buffer.clear();
269  }
270  else {
271  // Accumulate the next character into the sequence
272  buffer += character;
273  }
274  }
275 
276  // Add to the result if buffer still has contents or if the last character is a delimiter
277  if (!buffer.isEmpty() || string[string.getSize() - 1] == delimiter)
278  result.push_back(buffer);
279 
280  return result;
281  }
